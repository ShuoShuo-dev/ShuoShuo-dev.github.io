# 表单高级技巧

## 一、表单验证

### 1.1 HTML5内置验证

```html
<form>
    <!-- 必填 -->
    <input type="text" required>

    <!-- 最小/最大长度 -->
    <input type="text" minlength="3" maxlength="10">

    <!-- 数值范围 -->
    <input type="number" min="1" max="100">

    <!-- 数值步长 -->
    <input type="number" min="0" max="100" step="5">

    <!-- 邮箱 -->
    <input type="email">

    <!-- URL -->
    <input type="url">

    <!-- 电话（移动端显示数字键盘） -->
    <input type="tel">

    <!-- 正则验证 -->
    <input type="text" pattern="[A-Za-z]{3}" title="请输入3个字母">

    <!-- 日期 -->
    <input type="date">
    <input type="time">
    <input type="datetime-local">
    <input type="month">
    <input type="week">

    <!-- 颜色 -->
    <input type="color">

    <!-- 搜索 -->
    <input type="search">

    <button type="submit">提交</button>
</form>
```

### 1.2 自定义验证消息

```html
<input type="text" id="username"
       required minlength="3"
       oninvalid="this.setCustomValidity('用户名至少3个字符')"
       oninput="this.setCustomValidity('')">

<script>
document.getElementById('username').addEventListener('invalid', function(e) {
    if (this.validity.valueMissing) {
        this.setCustomValidity('请输入用户名');
    } else if (this.validity.tooShort) {
        this.setCustomValidity('用户名至少3个字符');
    } else {
        this.setCustomValidity('');
    }
});
</script>
```

### 1.3 CSS验证样式

```css
/* 有效状态 */
input:valid {
    border-color: green;
}

/* 无效状态 */
input:invalid {
    border-color: red;
}

/* 焦点时的无效状态 */
input:invalid:focus {
    border-color: red;
    box-shadow: 0 0 5px rgba(255, 0, 0, 0.3);
}

/* 悬停时的无效状态 */
input:not(:placeholder-shown):invalid {
    border-color: red;
}
```

## 四、复杂表单模式

### 4.1 级联选择

```html
<label for="province">省份：</label>
<select id="province">
    <option value="">请选择省份</option>
</select>

<label for="city">城市：</label>
<select id="city">
    <option value="">请先选择省份</option>
</select>

<script>
const data = {
    '北京': ['北京市'],
    '广东': ['广州', '深圳', '东莞', '佛山'],
    '浙江': ['杭州', '宁波', '温州', '绍兴']
};

const provinceSelect = document.getElementById('province');
const citySelect = document.getElementById('city');

// 初始化省份
Object.keys(data).forEach(province => {
    const option = document.createElement('option');
    option.value = province;
    option.textContent = province;
    provinceSelect.appendChild(option);
});

// 省份变更时更新城市
provinceSelect.addEventListener('change', function() {
    const cities = data[this.value] || [];
    citySelect.innerHTML = '<option value="">请选择城市</option>';
    cities.forEach(city => {
        const option = document.createElement('option');
        option.value = city;
        option.textContent = city;
        citySelect.appendChild(option);
    });
});
</script>
```

### 4.2 动态添加表单项

```html
<div id="skills-container">
    <div class="skill-item">
        <input type="text" name="skills[]" placeholder="技能">
        <button type="button" onclick="removeSkill(this)">删除</button>
    </div>
</div>
<button type="button" onclick="addSkill()">添加技能</button>

<script>
function addSkill() {
    const container = document.getElementById('skills-container');
    const newItem = document.createElement('div');
    newItem.className = 'skill-item';
    newItem.innerHTML = `
        <input type="text" name="skills[]" placeholder="技能">
        <button type="button" onclick="removeSkill(this)">删除</button>
    `;
    container.appendChild(newItem);
}

function removeSkill(btn) {
    const container = document.getElementById('skills-container');
    if (container.children.length > 1) {
        btn.parentElement.remove();
    } else {
        alert('至少保留一个技能项');
    }
}
</script>
```

### 4.3 文件上传增强

```html
<!-- 基础文件上传 -->
<input type="file">

<!-- 限制文件类型 -->
<input type="file" accept=".jpg,.png,.gif">
<input type="file" accept="image/*">
<input type="file" accept="video/*">
<input type="file" accept="audio/*">
<input type="file" accept=".pdf,.doc,.docx">

<!-- 多文件上传 -->
<input type="file" multiple>

<!-- 限制文件大小（需要JS配合） -->
<input type="file" id="fileInput" accept="image/*">
<span id="fileSize"></span>

<script>
document.getElementById('fileInput').addEventListener('change', function() {
    const file = this.files[0];
    if (file) {
        const size = (file.size / 1024 / 1024).toFixed(2);
        document.getElementById('fileSize').textContent = `文件大小: ${size} MB`;
        if (size > 5) {
            alert('文件大小不能超过5MB');
            this.value = '';
        }
    }
});
</script>

<!-- 显示预览 -->
<input type="file" id="avatar" accept="image/*" onchange="preview(this)">
<img id="preview" style="max-width: 200px;">

<script>
function preview(input) {
    if (input.files && input.files[0]) {
        const reader = new FileReader();
        reader.onload = function(e) {
            document.getElementById('preview').src = e.target.result;
        };
        reader.readAsDataURL(input.files[0]);
    }
}
</script>
```

## 五、表单安全

### 5.1 CSRF 防护

```html
<!-- 在表单中添加CSRF令牌 -->
<form action="/submit" method="POST">
    <input type="hidden" name="csrf_token" value="随机生成的令牌">
    <!-- 其他表单字段 -->
</form>
```

### 5.2 XSS 防护

```javascript
// 输入时转义
function escapeHtml(text) {
    const div = document.createElement('div');
    div.textContent = text;
    return div.innerHTML;
}

// 在提交前验证
form.addEventListener('submit', function(e) {
    const input = document.getElementById('comment');
    if (input.value.includes('<script>')) {
        e.preventDefault();
        alert('输入包含非法内容');
    }
});
```

### 5.3 密码安全

```html
<!-- 显示/隐藏密码 -->
<div class="password-field">
    <input type="password" id="password" name="password"
           minlength="8" required>
    <button type="button" onclick="togglePassword()">显示</button>
</div>

<script>
function togglePassword() {
    const input = document.getElementById('password');
    const btn = event.target;
    if (input.type === 'password') {
        input.type = 'text';
        btn.textContent = '隐藏';
    } else {
        input.type = 'password';
        btn.textContent = '显示';
    }
}
</script>
```

## 七、无障碍表单

### 7.1 ARIA属性

```html
<!-- 必填标记 -->
<label for="name">用户名 <span aria-hidden="true">*</span></label>
<input type="text" id="name" required aria-required="true">

<!-- 错误提示关联 -->
<label for="email">邮箱</label>
<input type="email" id="email" aria-describedby="email-help email-error">
<span id="email-help">请输入常用邮箱</span>
<span id="email-error" class="error" hidden>邮箱格式不正确</span>

<!-- 分组描述 -->
<fieldset>
    <legend>联系方式 <span aria-hidden="true">*</span></legend>
    <p id="contact-desc">请填写至少一种联系方式</p>

    <label for="phone">电话：</label>
    <input type="tel" id="phone" aria-describedby="contact-desc">
</fieldset>
```

### 7.2 实时验证反馈

```html
<label for="username">用户名</label>
<input type="text" id="username"
       aria-describedby="username-status"
       onkeyup="checkUsername(this)">
<span id="username-status" aria-live="polite"></span>

<script>
function checkUsername(input) {
    const status = document.getElementById('username-status');
    if (input.value.length < 3) {
        status.textContent = '用户名至少3个字符';
        status.className = 'error';
        input.setAttribute('aria-invalid', 'true');
    } else {
        status.textContent = '✓ 用户名可用';
        status.className = 'success';
        input.setAttribute('aria-invalid', 'false');
    }
}
</script>
```
