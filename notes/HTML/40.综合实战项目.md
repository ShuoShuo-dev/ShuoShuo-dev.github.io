function handleFormSubmit(e) {
    e.preventDefault();

    <div class="container">
        <h1>📝 待办事项</h1>

        <div class="input-group">
            <input type="text" id="todo-input" placeholder="添加新任务...">
            <button onclick="addTodo()">添加</button>
        </div>

        <div class="filters">
            <button class="filter-btn active" onclick="filterTodos('all')">全部</button>
            <button class="filter-btn" onclick="filterTodos('active')">未完成</button>
            <button class="filter-btn" onclick="filterTodos('completed')">已完成</button>
        </div>

        <ul class="todo-list" id="todo-list"></ul>

        <div class="count" id="count"></div>
    </div>

    <script>
        let todos = JSON.parse(localStorage.getItem('todos')) || [];
        let currentFilter = 'all';

        function saveTodos() {
            localStorage.setItem('todos', JSON.stringify(todos));
        }

        function renderTodos() {
            const list = document.getElementById('todo-list');
            list.innerHTML = '';

            const filtered = todos.filter(todo => {
                if (currentFilter === 'active') return !todo.completed;
                if (currentFilter === 'completed') return todo.completed;
                return true;
            });

            if (filtered.length === 0) {
                list.innerHTML = '<li class="empty">暂无任务</li>';
            }

            filtered.forEach((todo, index) => {
                const li = document.createElement('li');
                li.className = `todo-item ${todo.completed ? 'completed' : ''}`;
                li.innerHTML = `
                    <input type="checkbox" ${todo.completed ? 'checked' : ''}
                           onchange="toggleTodo(${index})">
                    <span class="text">${todo.text}</span>
                    <button class="delete-btn" onclick="deleteTodo(${index})">删除</button>
                `;
                list.appendChild(li);
            });

            updateCount();
        }

        function addTodo() {
            const input = document.getElementById('todo-input');
            const text = input.value.trim();

            if (text) {
                todos.push({ text, completed: false });
                saveTodos();
                renderTodos();
                input.value = '';
            }
        }

        function toggleTodo(index) {
            todos[index].completed = !todos[index].completed;
            saveTodos();
            renderTodos();
        }

        function deleteTodo(index) {
            todos.splice(index, 1);
            saveTodos();
            renderTodos();
        }

        function filterTodos(filter) {
            currentFilter = filter;
            document.querySelectorAll('.filter-btn').forEach(btn => {
                btn.classList.remove('active');
            });
            event.target.classList.add('active');
            renderTodos();
        }

        function updateCount() {
            const active = todos.filter(t => !t.completed).length;
            const completed = todos.length - active;
            document.getElementById('count').textContent =
                `共 ${todos.length} 项 | 未完成 ${active} | 已完成 ${completed}`;
        }

        // 回车添加
        document.getElementById('todo-input').addEventListener('keypress', e => {
            if (e.key === 'Enter') addTodo();
        });

        // 初始渲染
        renderTodos();
    </script>
</body>
</html>

## 三、开发流程建议

### 3.1 项目结构

```
project/
├── index.html
├── css/
│   ├── reset.css
│   ├── main.css
│   └── responsive.css
├── js/
│   ├── main.js
│   ├── components.js
│   └── utils.js
├── images/
│   └── ...
└── assets/
    └── ...
```

### 3.2 开发步骤

1. **规划阶段**
   - 明确项目需求
   - 绘制页面草图
   - 制定技术方案

2. **HTML阶段**
   - 创建语义化HTML结构
   - 添加必要的meta标签
   - 验证HTML有效性

3. **CSS阶段**
   - 重置默认样式
   - 编写基础样式
   - 添加响应式样式
   - 优化动画效果

4. **JavaScript阶段**
   - 实现交互功能
   - 处理表单验证
   - 优化性能
   - 添加错误处理

5. **测试阶段**
   - 多浏览器测试
   - 响应式测试
   - 性能测试
   - 可访问性检查

### 3.3 代码规范

- 使用语义化命名
- 保持代码整洁
- 添加必要注释
- 使用版本控制
- 编写可维护代码
