# JavaScript基础

## 一、JavaScript简介

JavaScript是一种脚本语言，用于网页的**交互和动态效果**。

### 1.1 HTML、CSS、JavaScript的关系

```
HTML - 结构
CSS  - 样式
JavaScript - 行为
```

### 1.2 JavaScript的引入方式

```html
<!-- 1. 行内脚本（不推荐） -->
<button onclick="alert('Hello!')">点击</button>

<!-- 2. 内部脚本 -->
<script>
    console.log('Hello World!');
</script>

<!-- 3. 外部脚本（推荐） -->
<script src="app.js"></script>
```

## 二、基础语法

### 2.1 变量声明

```javascript
// var - 函数作用域（不推荐）
var name = '张三';

// let - 块级作用域（推荐用于会改变的值）
let age = 18;
age = 20; // 可以重新赋值

// const - 常量（推荐用于不改变的值）
const PI = 3.14159;
// PI = 3; // 错误！常量不能重新赋值

// 声明多个变量
let x = 1, y = 2, z = 3;
```

### 2.2 数据类型

```javascript
// 原始类型
let str = "字符串";        // 字符串
let num = 123;            // 数字
let bool = true;          // 布尔值
let n = null;             // null
let u = undefined;        // undefined
let sym = Symbol('id');   // Symbol
let big = 123n;           // BigInt

// 引用类型
let arr = [1, 2, 3];      // 数组
let obj = { name: '张三' }; // 对象
let func = function() {}; // 函数

// 类型检测
console.log(typeof str);  // "string"
console.log(Array.isArray(arr)); // true
```

### 2.3 运算符

```javascript
// 算术运算符
let a = 10, b = 3;
console.log(a + b);  // 13
console.log(a - b);  // 7
console.log(a * b);  // 30
console.log(a / b);  // 3.333...
console.log(a % b);  // 1  (取余)
console.log(a ** b); // 1000 (幂)

// 赋值运算符
let x = 10;
x += 5;  // 15
x -= 3;  // 12
x *= 2;  // 24

// 比较运算符
console.log(5 == "5");   // true (松散相等)
console.log(5 === "5");  // false (严格相等)
console.log(5 != "5");   // false
console.log(5 !== "5");  // true

// 逻辑运算符
console.log(true && false);  // false
console.log(true || false);  // true
console.log(!true);          // false
```

## 三、DOM操作

### 3.1 选择元素

```javascript
// 通过ID
const header = document.getElementById('header');

// 通过类名（返回集合）
const items = document.getElementsByClassName('item');

// 通过标签名（返回集合）
const paragraphs = document.getElementsByTagName('p');

// CSS选择器（返回第一个）
const btn = document.querySelector('.btn');

// CSS选择器（返回所有）
const listItems = document.querySelectorAll('ul li');
```

### 3.2 修改元素内容

```javascript
const div = document.getElementById('content');

// 修改文本
div.textContent = '新的文本内容';

// 修改HTML
div.innerHTML = '<p>带有<strong>HTML</strong>的内容</p>';

// 修改属性
div.setAttribute('data-id', '123');
div.getAttribute('data-id');
div.removeAttribute('data-id');

// 直接操作属性
div.id = 'newId';
div.className = 'active';
div.style.color = 'red';
div.style.fontSize = '20px';
```

### 3.3 创建和添加元素

```javascript
// 创建元素
const newDiv = document.createElement('div');
newDiv.textContent = '新创建的div';
newDiv.className = 'card';

// 添加到页面
document.body.appendChild(newDiv);

// 插入到指定位置
parent.insertBefore(newDiv, referenceNode);
parent.insertAdjacentHTML('beforeend', '<p>HTML字符串</p>');
// beforebegin - 在元素前面
// afterbegin - 在元素内部开头
// beforeend - 在元素内部末尾
// afterend - 在元素后面

// 删除元素
parent.removeChild(child);
element.remove();
```

### 3.4 操作类名

```javascript
const box = document.querySelector('.box');

// 添加类
box.classList.add('active');

// 删除类
box.classList.remove('hidden');

// 切换类（有则删除，无则添加）
box.classList.toggle('active');

// 检查类
if (box.classList.contains('active')) {
    console.log('有active类');
}

// 替换类
box.classList.replace('old-class', 'new-class');
```

## 四、事件处理

### 4.1 事件监听

```javascript
const btn = document.getElementById('btn');

// addEventListener（推荐）
btn.addEventListener('click', function(event) {
    console.log('点击了！');
    console.log(event.target);
});

// 箭头函数
btn.addEventListener('click', (e) => {
    console.log(e.clientX, e.clientY);
});

// 移除事件
function handleClick() {
    console.log('点击');
}
btn.addEventListener('click', handleClick);
btn.removeEventListener('click', handleClick);

// 事件委托
document.querySelector('ul').addEventListener('click', function(e) {
    if (e.target.tagName === 'LI') {
        e.target.style.color = 'red';
    }
});
```

### 4.2 常见事件

```javascript
// 鼠标事件
element.addEventListener('click', handler);
element.addEventListener('dblclick', handler);
element.addEventListener('mouseenter', handler);  // 进入
element.addEventListener('mouseleave', handler);  // 离开
element.addEventListener('mousemove', handler);  // 移动

// 键盘事件
document.addEventListener('keydown', (e) => {
    if (e.key === 'Enter') console.log('回车');
    if (e.ctrlKey && e.key === 's') {
        e.preventDefault();
        console.log('Ctrl+S');
    }
});

// 表单事件
form.addEventListener('submit', (e) => {
    e.preventDefault();
    console.log('提交了');
});
input.addEventListener('focus', handler);
input.addEventListener('blur', handler);
input.addEventListener('change', handler);    // 失焦且值改变
input.addEventListener('input', handler);     // 值改变立即触发

// 窗口事件
window.addEventListener('load', handler);      // 页面加载完成
window.addEventListener('resize', handler);   // 窗口大小改变
window.addEventListener('scroll', handler);    // 滚动
```

## 五、数组操作

### 5.1 创建和访问

```javascript
const arr = [1, 2, 3, 4, 5];

// 访问
console.log(arr[0]);        // 1
console.log(arr.length);    // 5

// 修改
arr[0] = 10;

// 添加
arr.push(6);      // 末尾添加
arr.unshift(0);   // 开头添加

// 删除
arr.pop();        // 末尾删除
arr.shift();      // 开头删除
```

### 5.2 常用方法

```javascript
const arr = [1, 2, 3, 4, 5];

// 遍历
arr.forEach((item, index) => console.log(index, item));

// 映射
const doubled = arr.map(x => x * 2);  // [2, 4, 6, 8, 10]

// 过滤
const evens = arr.filter(x => x % 2 === 0);  // [2, 4]

// 查找
const found = arr.find(x => x > 3);     // 4
const foundIndex = arr.findIndex(x => x > 3); // 3
const exists = arr.includes(3);         // true

// 合并
const merged = arr1.concat(arr2);
const spread = [...arr1, ...arr2];

// 排序
const sorted = arr.slice().sort((a, b) => a - b); // 数值排序

// 截取
const sub = arr.slice(1, 3);  // [2, 3]

// 转为字符串
arr.join('-');  // "1-2-3-4-5"

// 判断
const allPositive = arr.every(x => x > 0);  // true
const hasNegative = arr.some(x => x < 0);  // false

// 累加
const sum = arr.reduce((acc, cur) => acc + cur, 0);  // 15
```

## 六、对象操作

### 6.1 创建和访问

```javascript
const person = {
    name: '张三',
    age: 18,
    sayHello() {
        console.log(`你好，我是${this.name}`);
    }
};

// 访问属性
console.log(person.name);
console.log(person['age']);

// 修改属性
person.age = 20;

// 添加属性
person.gender = '男';

// 删除属性
delete person.age;

// 检查属性
console.log('name' in person);      // true
console.log(person.hasOwnProperty('age')); // false
```

### 6.2 解构赋值

```javascript
const { name, age } = person;
const { name: userName, address: { city } } = person;

// 数组解构
const [first, second, ...rest] = [1, 2, 3, 4, 5];
console.log(first);  // 1
console.log(rest);   // [3, 4, 5]
```

## 七、字符串操作

### 7.1 常用方法

```javascript
const str = 'Hello World';

// 访问
str[0];              // "H"
str.charAt(0);       // "H"
str.substring(0, 5); // "Hello"
str.slice(-5);        // "World"

// 查找
str.indexOf('o');           // 4
str.lastIndexOf('o');       // 7
str.includes('World');     // true
str.startsWith('Hello');    // true
str.endsWith('!');          // false

// 转换
str.toUpperCase();   // "HELLO WORLD"
str.toLowerCase();   // "hello world"
str.trim();          // 去除首尾空格

// 替换
str.replace('World', 'JavaScript');  // "Hello JavaScript"
str.replace(/world/i, 'JS');          // 正则替换

// 分割
str.split(' ');     // ["Hello", "World"]
str.split('');      // ["H","e","l","l","o"," ","W","o","r","l","d"]

// 模板字符串
const name = '张三';
const message = `你好，${name}！今天是${new Date().getDay()}号。`;
```

## 八、异步操作

### 8.1 Promise

```javascript
const promise = new Promise((resolve, reject) => {
    setTimeout(() => {
        resolve('成功');
        // reject('失败');
    }, 1000);
});

promise
    .then(result => console.log(result))
    .catch(error => console.error(error))
    .finally(() => console.log('完成'));

// 多个Promise
Promise.all([promise1, promise2])
    .then(([result1, result2]) => console.log(result1, result2));

Promise.race([promise1, promise2])
    .then(result => console.log(result)); // 第一个完成的
```

### 8.2 async/await

```javascript
async function fetchData() {
    try {
        const response = await fetch('https://api.example.com/data');
        const data = await response.json();
        return data;
    } catch (error) {
        console.error('获取数据失败:', error);
    }
}

const data = await fetchData();
```

## 九、常用DOM API

### 9.1 类操作工具函数

```javascript
// 切换类
function toggleClass(element, className) {
    element.classList.toggle(className);
}

// 元素显示/隐藏
function show(element) {
    element.style.display = '';
}
function hide(element) {
    element.style.display = 'none';
}
function toggle(element) {
    element.style.display = element.style.display === 'none' ? '' : 'none';
}

// 检查是否包含某类
function hasClass(element, className) {
    return element.classList.contains(className);
}
```

### 9.2 动画辅助

```javascript
// 平滑滚动到顶部
function scrollToTop() {
    window.scrollTo({ top: 0, behavior: 'smooth' });
}

// 平滑滚动到元素
function scrollToElement(element) {
    element.scrollIntoView({ behavior: 'smooth' });
}

// 获取元素位置
function getPosition(element) {
    const rect = element.getBoundingClientRect();
    return {
        top: rect.top,
        left: rect.left,
        width: rect.width,
        height: rect.height
    };
}
```
