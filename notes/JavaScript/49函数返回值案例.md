## return 的两个作用

```javascript
// 1. 返回结果
function add(a, b) {
    return a + b;
}
let result = add(10, 20);
console.log(result);  // 30

// 2. 终止函数
function fn() {
    return 100;
    console.log('不会执行');  // 不执行
}
```

## 案例1：求最大值

```javascript
function getMax(arr) {
    let max = arr[0];
    for (let i = 1; i < arr.length; i++) {
        if (arr[i] > max) {
            max = arr[i];
        }
    }
    return max;
}

console.log(getMax([3, 5, 2, 9, 1]));  // 9
```

## 案例2：求最小值

```javascript
function getMin(arr) {
    let min = arr[0];
    for (let i = 1; i < arr.length; i++) {
        if (arr[i] < min) {
            min = arr[i];
        }
    }
    return min;
}

console.log(getMin([3, 5, 2, 9, 1]));  // 1
```

## 案例3：交换两数

```javascript
function swap(a, b) {
    return [b, a];  // 返回数组
}

let arr = swap(10, 20);
console.log(arr);  // [20, 10]
```

## arguments 对象

```javascript
// 实参个数不固定时
function getSum() {
    // arguments 存储所有实参
    let sum = 0;
    for (let i = 0; i < arguments.length; i++) {
        sum += arguments[i];
    }
    return sum;
}

console.log(getSum(1, 2, 3));      // 6
console.log(getSum(1, 2, 3, 4, 5)); // 15
```

## 注意事项

1. **return 只能返回一个值**
2. **return 后代码不执行**
3. **arguments 是伪数组**：有 length，能遍历

## 记忆口诀

> **返回值**：return 带结果，多值返回用数组，arguments 接收所有实参
