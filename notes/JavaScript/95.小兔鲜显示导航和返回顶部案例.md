# 95.小兔鲜显示导航和返回顶部案例

## 一、效果演示

### 功能需求

1. **滚动超过一定距离**：顶部导航栏固定显示
2. **点击返回顶部**：平滑滚动回到页面顶部
3. **滚动进度条**：实时显示页面滚动进度

### 效果预览

```
┌─────────────────────────────────┐
│  [========进度条========]       │  ← 滚动时显示进度
├─────────────────────────────────┤
│                                 │
│         页面内容区域             │
│                                 │
│                                 │
│                        [↑]      │  ← 滚动后显示返回顶部按钮
└─────────────────────────────────┘
```

## 四、JavaScript 实现

```javascript
// ========================================
// 获取页面元素
// ========================================
const header = document.getElementById('header');           // 顶部导航栏
const backTop = document.getElementById('backTop');         // 返回顶部按钮
const progressBar = document.getElementById('progressBar'); // 滚动进度条

// ========================================
// 配置参数
// ========================================
const SCROLL_THRESHOLD = 500;  // 滚动超过 500px 显示导航和返回顶部
const ANIMATION_DURATION = 500; // 返回顶部动画时长（ms）

// ========================================
// 节流函数（性能优化）
// ========================================
function throttle(func, wait) {
    let previous = 0;
    let timeout = null;

    return function() {
        const now = Date.now();
        const remaining = wait - (now - previous);

        if (remaining <= 0 || remaining > wait) {
            if (timeout) {
                clearTimeout(timeout);
                timeout = null;
            }
            previous = now;
            func.apply(this, arguments);
        } else if (!timeout) {
            timeout = setTimeout(() => {
                previous = Date.now();
                timeout = null;
                func.apply(this, arguments);
            }, remaining);
        }
    };
}

// ========================================
// 滚动事件处理
// ========================================
function handleScroll() {
    const scrollY = window.scrollY;

    // 1. 控制导航栏显示/隐藏
    if (scrollY > SCROLL_THRESHOLD) {
        header.classList.add('show');
    } else {
        header.classList.remove('show');
    }

    // 2. 控制返回顶部按钮显示/隐藏
    if (scrollY > SCROLL_THRESHOLD) {
        backTop.classList.add('show');
    } else {
        backTop.classList.remove('show');
    }

    // 3. 更新滚动进度条
    updateProgressBar();
}

// ========================================
// 更新滚动进度条
// ========================================
function updateProgressBar() {
    // 获取文档总高度和视口高度
    const docHeight = document.documentElement.scrollHeight;
    const viewHeight = window.innerHeight;

    // 计算最大可滚动距离
    const maxScroll = docHeight - viewHeight;

    // 计算滚动百分比
    const scrollPercent = maxScroll > 0
        ? (window.scrollY / maxScroll) * 100
        : 0;

    // 更新进度条宽度
    progressBar.style.width = Math.min(scrollPercent, 100) + '%';
}

// ========================================
// 返回顶部功能
// ========================================
function scrollToTop() {
    // 获取当前滚动位置
    const startY = window.scrollY;

    // 记录开始时间
    const startTime = Date.now();

    // 动画函数
    function animate() {
        // 计算已用时间
        const elapsed = Date.now() - startTime;

        // 计算进度（0 ~ 1）
        const progress = Math.min(elapsed / ANIMATION_DURATION, 1);

        // 使用 easeOutQuart 缓动函数，让动画更自然
        // 公式：1 - (1 - x)^3
        const easeProgress = 1 - Math.pow(1 - progress, 3);

        // 计算当前应该滚动到的位置
        const currentY = startY * (1 - easeProgress);

        // 执行滚动
        window.scrollTo(0, currentY);

        // 如果动画未完成，继续执行
        if (progress < 1) {
            requestAnimationFrame(animate);
        }
    }

    // 启动动画
    requestAnimationFrame(animate);
}

// ========================================
// 绑定事件监听
// ========================================

// 1. 监听页面滚动（使用节流优化）
window.addEventListener('scroll', throttle(handleScroll, 50));

// 2. 监听返回顶部按钮点击
backTop.addEventListener('click', scrollToTop);

// 3. 页面加载时检查滚动位置（防止刷新后状态不一致）
window.addEventListener('load', () => {
    handleScroll();
});

// 4. 窗口大小改变时重新计算进度条
window.addEventListener('resize', throttle(updateProgressBar, 100));
```

## 五、优化版本（使用 class）

```javascript
class BackToTop {
    constructor(options = {}) {
        // 配置参数
        this.scrollThreshold = options.scrollThreshold || 500;
        this.animationDuration = options.animationDuration || 500;
        this.easingFunc = options.easingFunc || this.easeOutQuart;

        // 获取元素
        this.header = document.getElementById(options.headerId || 'header');
        this.backTop = document.getElementById(options.backTopId || 'backTop');
        this.progressBar = document.getElementById(options.progressBarId || 'progressBar');

        // 状态标志
        this.isAnimating = false;

        // 初始化
        this.init();
    }

    init() {
        // 绑定滚动事件
        this.bindScroll();

        // 绑定返回顶部点击事件
        this.bindBackTop();

        // 初始化状态
        this.handleScroll();
    }

    // 滚动事件处理
    bindScroll() {
        let timeout = null;

        window.addEventListener('scroll', () => {
            if (timeout) clearTimeout(timeout);

            timeout = setTimeout(() => {
                this.handleScroll();
            }, 50);
        });
    }

    // 返回顶部事件
    bindBackTop() {
        this.backTop.addEventListener('click', () => {
            this.animateToTop();
        });
    }

    // 处理滚动逻辑
    handleScroll() {
        const scrollY = window.scrollY;
        const isBeyond = scrollY > this.scrollThreshold;

        // 控制导航栏
        this.header.classList.toggle('show', isBeyond);

        // 控制返回顶部按钮
        this.backTop.classList.toggle('show', isBeyond);

        // 更新进度条
        this.updateProgress();
    }

    // 更新进度条
    updateProgress() {
        const docHeight = document.documentElement.scrollHeight - window.innerHeight;
        const progress = docHeight > 0 ? (window.scrollY / docHeight) * 100 : 0;
        this.progressBar.style.width = Math.min(progress, 100) + '%';
    }

    // 返回顶部动画
    animateToTop() {
        if (this.isAnimating) return;

        this.isAnimating = true;
        const startY = window.scrollY;
        const startTime = Date.now();

        const animate = () => {
            const elapsed = Date.now() - startTime;
            const progress = Math.min(elapsed / this.animationDuration, 1);
            const easeProgress = this.easingFunc(progress);

            window.scrollTo(0, startY * (1 - easeProgress));

            if (progress < 1) {
                requestAnimationFrame(animate);
            } else {
                this.isAnimating = false;
            }
        };

        requestAnimationFrame(animate);
    }

    // 缓动函数 - 二次方缓出
    easeOutQuart(t) {
        return 1 - Math.pow(1 - t, 4);
    }
}

// 使用方式
document.addEventListener('DOMContentLoaded', () => {
    const backToTop = new BackToTop({
        scrollThreshold: 500,
        animationDuration: 500,
        easingFunc: t => 1 - Math.pow(1 - t, 3)  // 自定义缓动函数
    });
});
```

## 七、核心知识点总结

| 功能 | 实现方法 | 关键代码 |
|------|----------|----------|
| 固定导航 | CSS `position: fixed` + JS `transform` | `header.classList.toggle('show')` |
| 返回顶部 | `window.scrollTo(0, 0)` + 动画 | `requestAnimationFrame` + 缓动函数 |
| 进度条 | 计算滚动百分比 | `(scrollY / maxScroll) * 100` |
| 节流优化 | `setTimeout` 控制执行频率 | `50ms` 间隔 |
