# 93.阻止元素默认行为

## 什么是默认行为？

浏览器元素有一些自带的默认行为，比如：
- 点击 `<a>` 链接会跳转页面
- 点击提交按钮会提交表单
- 右键会弹出菜单
- 按住鼠标拖动会出现选中文本

## 阻止默认行为的方法

### 1. `preventDefault()`

最常用的方法，阻止元素的默认行为：

```javascript
// 阻止链接跳转
document.querySelector('a').addEventListener('click', function(e) {
    e.preventDefault();
    console.log('链接被阻止跳转了！');
});

// 阻止表单提交
document.querySelector('form').addEventListener('submit', function(e) {
    e.preventDefault();
    console.log('表单被阻止提交了！');
});

// 阻止右键菜单
document.addEventListener('contextmenu', function(e) {
    e.preventDefault();
    console.log('右键菜单被阻止了！');
});

// 阻止选中文本（双击或拖动）
document.addEventListener('selectstart', function(e) {
    e.preventDefault();
});
```

### 2. `return false`

在 jQuery 中，`return false` 可以阻止默认行为并阻止冒泡：

```javascript
// jQuery 写法
$('a').click(function() {
    console.log('阻止默认行为');
    return false; // 等同于 e.preventDefault() + e.stopPropagation()
});
```

### 3. `oncontextmenu` 等事件属性

直接设置事件属性返回 `false`：

```javascript
// 阻止右键
document.oncontextmenu = function() {
    return false;
};

// 阻止选中文本
document.onselectstart = function() {
    return false;
};
```

## 常见应用场景

### 1. 自定义链接行为

```javascript
const links = document.querySelectorAll('.custom-link');
links.forEach(link => {
    link.addEventListener('click', function(e) {
        e.preventDefault();
        const href = this.getAttribute('href');
        // 自定义跳转逻辑
        console.log('自定义处理:', href);
    });
});
```

### 2. 表单验证后提交

```javascript
const form = document.querySelector('form');

form.addEventListener('submit', function(e) {
    e.preventDefault();

    // 验证表单
    const username = document.getElementById('username').value;
    if (username.length < 3) {
        alert('用户名至少3个字符！');
        return;
    }

    // 验证通过，手动提交
    console.log('表单验证通过，提交数据...');
    // this.submit(); // 或者用 AJAX 提交
});
```

### 3. 自定义右键菜单

```javascript
const menu = document.querySelector('#custom-menu');

document.addEventListener('contextmenu', function(e) {
    e.preventDefault();

    // 显示自定义菜单
    menu.style.display = 'block';
    menu.style.left = e.pageX + 'px';
    menu.style.top = e.pageY + 'px';
});

// 点击其他地方隐藏菜单
document.addEventListener('click', function() {
    menu.style.display = 'none';
});
```

### 4. 实现拖拽功能时阻止默认行为

```javascript
const draggable = document.querySelector('.draggable');

draggable.addEventListener('mousedown', function(e) {
    e.preventDefault(); // 阻止默认选中文本行为
    // 开始拖拽逻辑...
});
```

## 完整示例：点击按钮不跳转

```html

<html lang="zh-CN">
<head>

            border: 1px solid #ddd;
        }
        input {
            padding: 8px;
            margin-right: 10px;
        }
    </style>
</head>
<body>
    <h2>阻止默认行为示例</h2>

    <a href="https://www.baidu.com" id="myLink">百度链接（被阻止跳转）</a>

    <form id="myForm">
        <input type="text" placeholder="输入内容">
        <button type="submit">提交表单</button>
    </form>

    <script>
        // 阻止链接跳转
        document.getElementById('myLink').addEventListener('click', function(e) {
            e.preventDefault();
            console.log('链接被阻止了！');
            alert('链接跳转被阻止了，点击确定手动跳转');
            // 可以手动跳转
            // window.location.href = 'https://www.baidu.com';
        });

        // 阻止表单提交
        document.getElementById('myForm').addEventListener('submit', function(e) {
            e.preventDefault();
            console.log('表单提交被阻止了！');

            const input = this.querySelector('input');
            if (input.value.trim() === '') {
                alert('请输入内容！');
                return;
            }

            alert('表单验证通过，可以提交了！');
        });
    </script>
</body>

```

## 注意事项

1. **`preventDefault()` vs `return false`**
   - `preventDefault()` 只能阻止默认行为，不阻止冒泡
   - `return false` 在 jQuery 中既能阻止默认行为，也能阻止冒泡
   - 原生 JS 中 `return false` 只阻止默认行为

2. **有些默认行为无法阻止**
   - 浏览器安全限制：地址栏输入 URL 无法阻止
   - 一些浏览器保护机制

3. **阻止默认行为后要提供替代方案**
   - 用户期望的行为被阻止后，应该有其他交互提示

## 小结

| 方法 | 适用范围 | 说明 |
|------|----------|------|
| `e.preventDefault()` | 原生 JS | 推荐使用，精确控制 |
| `return false` | jQuery | 同时阻止默认行为和冒泡 |
| `oncontextmenu = false` | 事件属性 | 简洁但不推荐 |
