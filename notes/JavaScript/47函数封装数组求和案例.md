## 需求：封装数组求和函数

```javascript
// 接收一个数组，返回所有元素之和
function getArraySum(arr) {
    // arr 是形参，接收传入的数组
    let sum = 0;
    for (let i = 0; i < arr.length; i++) {
        sum += arr[i];
    }
    return sum;
}

// 调用
let result = getArraySum([1, 2, 3, 4, 5]);
console.log(result);  // 15
```

## 带默认参数版本

```javascript
function getArraySum(arr = []) {
    let sum = 0;
    for (let i = 0; i < arr.length; i++) {
        sum += arr[i];
    }
    return sum;
}

console.log(getArraySum());        // 0
console.log(getArraySum([1,2,3])); // 6
```

## 封装累加和函数

```javascript
// 求 n 到 m 的累加和
function getSum(n = 0, m = 0) {
    let sum = 0;
    for (let i = n; i <= m; i++) {
        sum += i;
    }
    return sum;
}

// 调用
console.log(getSum(1, 100));    // 5050
console.log(getSum(50, 100));   // 3825
console.log(getSum());          // 0
```

## 实战：用户输入求和

```javascript
function getSum(n = 0, m = 0) {
    let sum = 0;
    for (let i = n; i <= m; i++) {
        sum += i;
    }
    return sum;
}

let start = +prompt('起始值：');
let end = +prompt('结束值：');
let result = getSum(start, end);
console.log(`从${start}到${end}的和是：${result}`);
```

## 注意事项

1. **数组是形参**：不传参会报错或为 undefined
2. **设置默认值**：防止不传参出错
3. **return 返回结果**：调用者拿到结果

> **数组求和**：函数参数接收数组，遍历累加得结果，return 带回函数外
